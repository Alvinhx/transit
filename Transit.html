<body>


    <div class="header">Capitol Hill Station - Route 8</div>

    <div id="board" class="board">

        <div class="row"><div class="dest">Loading Transit Data...</div></div>

    </div>


    <script>

        const API_KEY = "TEST"; // Public test key

        const STOPS = ["1_29270", "1_29262"]; // Bay 1 & Bay 2

        

        async function updateBoard() {

            let allArrivals = [];


            for (let stopId of STOPS) {

                try {

                    const response = await fetch(`https://api.pugetsound.onebusaway.org/api/where/arrivals-and-departures-for-stop/${stopId}.json?key=${API_KEY}`);

                    const json = await response.json();

                    const arrivals = json.data.entry.arrivalsAndDepartures;

                    

                    // Filter specifically for Route 8

                    const route8 = arrivals.filter(a => a.routeShortName === "8");

                    allArrivals = [...allArrivals, ...route8];

                } catch (e) {

                    console.error("Fetch failed", e);

                }

            }


            // Sort by arrival time

            allArrivals.sort((a, b) => (a.scheduledArrivalTime || a.predictedArrivalTime) - (b.scheduledArrivalTime || b.predictedArrivalTime));


            // Take top 4

            render(allArrivals.slice(0, 4));

        }


        function render(data) {

            const board = document.getElementById('board');

            if (data.length === 0) {

                board.innerHTML = '<div class="row"><div class="dest">No buses found</div></div>';

                return;

            }


            board.innerHTML = data.map(bus => {

                const arrivalTime = bus.predictedArrivalTime || bus.scheduledArrivalTime;

                const minutes = Math.round((arrivalTime - Date.now()) / 60000);

                const displayTime = minutes <= 0 ? "NOW" : minutes;

                

                return `

                    <div class="row">

                        <div class="route">${bus.routeShortName}</div>

                        <div class="dest">${bus.tripHeadsign.split(" - ")[0]}</div>

                        <div class="time">${displayTime}<span class="unit">${minutes > 0 ? 'MIN' : ''}</span></div>

                    </div>

                `;

            }).join('');

        }


        // Update every 30 seconds

        setInterval(updateBoard, 30000);

        updateBoard();

    </script>
